import statistics as stats

avg_mpg = stats.mean(mpg)
avg_weight = stats.mean(weight)
n = len(mpg)

def get_y(x):
    top = 0
    bottom = 0

    for i in range(n):
        top += (weight[i] * mpg[i]) 
        bottom += (weight[i]) ** 2 

    slope = (top - (n * avg_weight * avg_mpg)) / (bottom - (n * (avg_weight)**2))
    
    return avg_mpg + slope*(x - avg_weight)
